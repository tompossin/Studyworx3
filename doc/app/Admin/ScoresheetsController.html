def create
  @scoresheet = Scoresheet.new(params[:scoresheet])
  @scoresheet.school_id = @school.id
  @scoresheet.save
  respond_to do |format|
    format.html {redirect_to admin_school_scoresheets_path(@school)}
  end
end
