    class ProfilesController

    This is just a controller - it will be calling pages that edit the users table and the teams table.

    This will be the main controller for user info updates,etc. Most of these actions will render to the nav_bar via AJAX

    Profiles Controller

    This controller will give an understandable url and will make it easier to avoid route conflicts with devise.

def bio_save
  @note = current_user.note
  @note.bio = params[:note][:bio]
  @note.save
  @user = current_user
  respond_to do |format|
    if @note.save
      format.js
    else
      format.js {render "shared/save_failed"}
    end
  end
end

def public_load
  @note = Note.find_by_user_id(current_user.id)
  respond_to do |format|
    if @note
      format.js
    else
      @note = Note.new
      @note.user_id = current_user.id
      @note.public = "Click the edit link above to add your thought to share..."
      @note.save
      format.js
    end
  end 
end

def settheme
  current_user.preference.theme = params[:preference][:theme]
  current_user.preference.rows = params[:rows]
  current_user.preference.bgcolor = params[:preference][:bgcolor]
  
  respond_to do |format|
    if current_user.preference.save
      set_session
      format.js {render "reminder_save"}
    else
      format.js {render "shared/save_failed"}
    end
  end
end

def update
  if params[:id] == current_user.id.to_s
    @user = User.find(params[:id])
    @user.update_attributes!(params[:user])
  end
  redirect_to edit_profile_path(current_user.id)
end
